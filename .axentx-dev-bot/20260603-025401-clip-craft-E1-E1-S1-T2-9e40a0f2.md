// src/template_selector.js
export class TemplateSelector {
  /**
   * @param {HTMLElement} container - Where to render the selector.
   * @param {function(string):void} onApply - Called with the selected template ID when the user clicks Apply.
   */
  constructor(container, onApply) {
    this.container = container;
    this.onApply = onApply;
    this.templates = TemplateSelector.templates; // static data
    this._render();
  }

  _render() {
    // Load the template_selector.html fragment
    fetch('/public/template_selector.html')
      .then(r => r.text())
      .then(html => {
        this.container.innerHTML = html;
        this._populateOptions();
        this._attachEvents();
      });
  }

  _populateOptions() {
    const select = this.container.querySelector('#template-select');
    this.templates.forEach(t => {
      const opt = document.createElement('option');
      opt.value = t.id;
      opt.textContent = t.name;
      select.appendChild(opt);
    });
  }

  _attachEvents() {
    const btn = this.container.querySelector('#template-apply');
    btn.addEventListener('click', () => {
      const selected = this.container.querySelector('#template-select').value;
      if (this.onApply) this.onApply(selected);
    });
  }
}

  { id: 'tpl-02', name: 'Product Demo (4K, 60fps)' },
  { id: 'tpl-03', name: 'Social Media (720p, 30fps)' },
];
