-- Add analytics tables to track video creation and user engagement
CREATE TABLE video_analytics (
    id SERIAL PRIMARY KEY,
    video_id INT NOT NULL REFERENCES videos(id),
    creation_time TIMESTAMP WITH TIME ZONE NOT NULL,
    duration_seconds INT NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'draft',
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE user_engagement (
    id SERIAL PRIMARY KEY,
    video_id INT NOT NULL REFERENCES videos(id),
    user_id INT NOT NULL REFERENCES users(id),
    event_type VARCHAR(20) NOT NULL, -- 'view', 'like', 'comment', 'share'
    event_time TIMESTAMP WITH TIME ZONE NOT NULL,
    metadata JSONB, -- optional: comment text, share link, etc.
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
```

// /opt/axentx/clip-craft/src/analytics.js

const { Pool } = require('pg');
const pool = new Pool({
  // Database connection config (e.g., from environment variables)
  user: process.env.DB_USER,
  host: process.env.DB_HOST,
  database: process.env.DB_NAME,
  password: process.env.DB_PASSWORD,
  port: process.env.DB_PORT,
});

/**
 * Logs video creation metrics (start time, duration, status).
 * @param {number} videoId - ID of the created video.
 * @param {Date} startTime - Timestamp when video creation started.
 * @param {Date} endTime - Timestamp when video creation finished.
 */
async function logVideoCreation(videoId, startTime, endTime) {
  const duration = Math.round((endTime - startTime) / 1000); // seconds
  const query = `
    INSERT INTO video_analytics (video_id, creation_time, duration_seconds, status)
    VALUES ($1, $2, $3, 'draft')
    ON CONFLICT (video_id) DO UPDATE
    SET creation_time = EXCLUDED.creation_time,
        duration_seconds = EXCLUDED.duration_seconds,
        status = EXCLUDED.status
  `;
  await pool.query(query, [videoId, startTime, duration]);
}

/**
 * Logs user engagement events (views, likes, comments, shares).
 * @param {number} videoId - ID of the video.
 * @param {number} userId - ID of the user.
 * @param {string} eventType - Type of event (e.g., 'view', 'like').
 * @param {Object} metadata - Optional event details (e.g., comment text).
 */
async function logUserEngagement(videoId, userId, eventType, metadata = {}) {
  const query = `
    INSERT INTO user_engagement (video_id, user_id, event_type, event_time, metadata)
    VALUES ($1, $2, $3, $4, $5)
    ON CONFLICT (video_id, user_id, event_type, event_time) DO NOTHING
  `;
  await pool.query(query, [videoId, userId, eventType, new Date(), JSON.stringify(metadata)]);
}

module.exports = {
  logVideoCreation,
  logUserEngagement,
};
```
