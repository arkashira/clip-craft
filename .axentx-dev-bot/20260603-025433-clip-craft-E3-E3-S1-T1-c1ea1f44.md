### 1. Database Schema (`db_schema.sql`)

```sql
-- analytics table – stores per‑video metrics
CREATE TABLE IF NOT EXISTS analytics (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    video_id    UUID NOT NULL REFERENCES videos(id) ON DELETE CASCADE,
    creation_ts TIMESTAMP WITH TIME ZONE NOT NULL,
    views       BIGINT DEFAULT 0,
    likes       BIGINT DEFAULT 0,
    shares      BIGINT DEFAULT 0,
    created_at  TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Index for fast filtering by video and time
CREATE INDEX IF NOT EXISTS idx_analytics_video_ts
    ON analytics (video_id, created_at);

-- Trigger to keep updated_at current
CREATE OR REPLACE FUNCTION analytics_update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_analytics_timestamp ON analytics;
CREATE TRIGGER trg_analytics_timestamp
    BEFORE UPDATE ON analytics
    FOR EACH ROW
    EXECUTE FUNCTION analytics_update_timestamp();
```

> **Why this works**  
> * `ON DELETE CASCADE` keeps the analytics clean when a video is removed.  
> * The composite index supports queries like “top videos in the last week”.  
> * The trigger guarantees `updated_at` is always accurate without manual updates.
