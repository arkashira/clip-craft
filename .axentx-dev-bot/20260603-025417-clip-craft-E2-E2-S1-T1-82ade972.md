3. **Playback Controls** – The video element exposes native controls (`controls` attribute) and can be played, paused, and seeked.  
4. **Quality Validation** – On load, the video’s resolution is ≥ 720p and frame rate ≥ 30fps.  
5. **Refinement Trigger** – A “Refine” button (`#refine-btn`) is enabled only after the video has successfully loaded and passes quality checks.  
6. **Error Handling** – If the video fails to load or does not meet quality standards, an error banner (`#error-banner`) is displayed and the “Refine” button is disabled.

// video_preview.test.js
import { render, screen, fireEvent } from '@testing-library/dom';
import { initVideoPreview } from '../src/video_preview.js';

describe('Video Preview Component', () => {
  beforeEach(() => {
    document.body.innerHTML = `
      <video id="preview-video" controls></video>
      <button id="refine-btn" disabled>Refine</button>
      <div id="error-banner" hidden></div>
    `;
  });

  test('renders video element with correct id', () => {
    const video = screen.getByTestId('preview-video');
    expect(video).toBeInTheDocument();
  });

  test('sets src to generated video URL', () => {
    initVideoPreview('/generated/123.mp4');
    const video = screen.getByTestId('preview-video');
    expect(video.src).toContain('/generated/123.mp4');
  });

  test('enables refine button after successful load', async () => {
    initVideoPreview('/generated/123.mp4');
    const video = screen.getByTestId('preview-video');
    const refineBtn = screen.getByTestId('refine-btn');

    // Simulate successful load event
    fireEvent.loadedmetadata(video);
    await Promise.resolve(); // wait for async handlers

    expect(refineBtn).not.toBeDisabled();
  });

  test('displays error banner if video fails to load', async () => {
    initVideoPreview('/generated/invalid.mp4');
    const video = screen.getByTestId('preview-video');
    const errorBanner = screen.getByTestId('error-banner');
    const refineBtn = screen.getByTestId('refine-btn');

    fireEvent.error(video);
    await Promise.resolve();

    expect(errorBanner).not.toBeHidden();
    expect(refineBtn).toBeDisabled();
  });

  test('validates resolution and frame rate', async () => {
    initVideoPreview('/generated/123.mp4');
    const video = screen.getByTestId('preview-video');

    // Mock video metadata

});
