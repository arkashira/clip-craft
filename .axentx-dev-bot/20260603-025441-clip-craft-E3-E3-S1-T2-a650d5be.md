// Import required modules
const express = require('express');
const mongoose = require('mongoose');

// Initialize Express app
const app = express();

// Database connection
mongoose.connect('mongodb://localhost/axentx', { useNewUrlParser: true, useUnifiedTopology: true });

// Define the analytics data schema
const analyticsSchema = new mongoose.Schema({
  videoId: String,
  creationTime: Date,
  engagement: Number
});

// Create Mongoose model
const AnalyticsData = mongoose.model('AnalyticsData', analyticsSchema);

// API endpoint to fetch analytics data
app.get('/analytics', async (req, res) => {
  try {
    let query = {};
    if (req.query.videoId) query.videoId = req.query.videoId;
    if (req.query.creationTime) query.creationTime = req.query.creationTime;
    if (req.query.engagement) query.engagement = req.query.engagement;

    const analyticsData = await AnalyticsData.find(query).sort(req.query.sort || '-creationTime');
    res.json(analyticsData);
  } catch (error) {
    console.error(error);
    res.status(500).json({ message: 'Error fetching analytics data' });
  }
});

// Start the server
const port = 3001;
app.listen(port, () => {
  console.log(`Server listening on port ${port}`);
});
```
