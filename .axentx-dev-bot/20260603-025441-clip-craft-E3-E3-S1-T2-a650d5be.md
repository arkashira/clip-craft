    // ---------- Chart Setup ----------
    const ctx = document.getElementById('analytics-chart').getContext('2d');
    const analyticsChart = new Chart(ctx, {
      type: 'bar',
      data: {
        labels: [],          // dates
        datasets: [{
          label: 'Visits',
          data: [],          // values
          backgroundColor: 'rgba(75, 192, 192, 0.2)',
          borderColor: 'rgba(75, 192, 192, 1)',
          borderWidth: 1
        }]
      },
      options: {
        responsive: true,
        scales: {
          y: { beginAtZero: true }
        }
      }
    });

    // ---------- Helper Functions ----------
    const showError = (msg) => {
      const el = document.getElementById('error-message');
      el.textContent = msg;
      el.style.display = msg ? 'block' : 'none';
    };

    const fetchAnalytics = async (filter) => {
      try {
        const resp = await fetch(`/api/analytics?filter=${filter}`);
        if (!resp.ok) throw new Error(`Status ${resp.status}`);
        const data = await resp.json();
        return data;
      } catch (e) {
        showError(`Failed to load data: ${e.message}`);
        return null;
      }
    };

    const renderChart = (data) => {
      analyticsChart.data.labels = data.map(i => i.date);
      analyticsChart.data.datasets[0].data = data.map(i => i.value);
      analyticsChart.update();
    };

    const sortData = (data, order) => {
      return data.slice().sort((a, b) => {
        if (order === 'asc') return a.value - b.value;
        return b.value - a.value;
      });
    };

    // ---------- Event Handlers ----------
    const loadAndRender = async () => {
      const filter = document.getElementById('filter-select').value;
      const raw = await fetchAnalytics(filter);
      if (!raw) return;

      const order = document.getElementById('sort-button').dataset.order;
      const processed = sortData(raw, order);
      renderChart(processed);
    };

    document.getElementById('filter-select')
      .addEventListener('change', loadAndRender);

    document.getElementById('sort-button')
      .addEventListener('click', (e) => {
        const btn = e.target;
        btn.dataset.order = btn.dataset.order === 'asc' ? 'desc' : 'asc';
        btn.textContent = btn.dataset.order === 'asc' ? 'Sort ↑' : 'Sort ↓';
        loadAndRender();
      });

    // ---------- Initial Load ----------
    loadAndRender();
  </script>
