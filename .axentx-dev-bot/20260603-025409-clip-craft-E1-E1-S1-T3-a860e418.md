const express = require('express');
const router = express.Router();
const { generateVideo } = require('../video_generator');
const fs = require('fs');
const path = require('path');

// POST /api/generate
router.post('/generate', async (req, res) => {
  const { script, templateId } = req.body;
  if (!script || !templateId) {
    return res.status(400).json({ error: 'script and templateId required' });
  }

  try {
    const videoPath = await generateVideo(script, templateId);
    // Stream the video back
    res.setHeader('Content-Type', 'video/mp4');
    const readStream = fs.createReadStream(videoPath);
    readStream.pipe(res);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: 'Video generation failed' });
  }
});

module.exports = router;
```

```js
const { spawn } = require('child_process');
const fs = require('fs');
const os = require('os');
const path = require('path');
const { v4: uuidv4 } = require('uuid');

/**
 * generateVideo
 * @param {string} script - Markdown/text script
 * @param {string} templateId - ID of the chosen template
 * @returns {Promise<string>} path to generated MP4
 */
async function generateVideo(script, templateId) {
  // 1. Write script to temp file
  const scriptPath = path.join(os.tmpdir(), `script-${uuidv4()}.md`);
  fs.writeFileSync(scriptPath, script, 'utf8');

  // 2. Prepare output path
  const outPath = path.join(os.tmpdir(), `video-${uuidv4()}.mp4`);

  // 3. Call Python model
  const py = spawn('python3', ['ml_model.py', scriptPath, templateId, outPath]);

  return new Promise((resolve, reject) => {
    py.on('error', reject);
    py.on('close', (code) => {
      if (code !== 0) {
        return reject(new Error(`ml_model exited with ${code}`));
      }
      // 4. Return output path
      resolve(outPath);
    });
  });
}

module.exports = { generateVideo };
```
