// server.js
const express = require('express');
const path = require('path');
const fs = require('fs');
const multer = require('multer');
const { generateVideo } = require('./video_generator');

const app = express();
app.use(express.json());

// -------------------------------------------------
// Multer – in‑memory upload, 5 MB limit, markdown/text only
// -------------------------------------------------
const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 5 * 1024 * 1024 }, // 5 MB
  fileFilter: (req, file, cb) => {
    const allowed = ['text/plain', 'text/markdown', 'application/markdown'];
    if (!allowed.includes(file.mimetype)) {
      return cb(new Error('Only Markdown or plain‑text files are allowed'));
    }
    cb(null, true);
  },
});

// -------------------------------------------------
// POST /api/generate-video
// -------------------------------------------------
app.post('/api/generate-video', upload.single('script'), async (req, res) => {
  try {
    const { templateId } = req.body;
    if (!templateId) {
      return res.status(400).json({ error: 'templateId is required' });
    }
    if (!req.file) {
      return res.status(400).json({ error: 'script file is required' });
    }

    // ---- 1️⃣ Save uploaded script to a temporary file ----
    const tmpDir = path.join(__dirname, 'tmp');
    await fs.promises.mkdir(tmpDir, { recursive: true });
    const scriptPath = path.join(
      tmpDir,
      `${Date.now()}_${req.file.originalname}`
    );
    await fs.promises.writeFile(scriptPath, req.file.buffer);

    // ---- 2️⃣ Call the Python model via the helper ----
    const outputPath = await generateVideo(
      scriptPath,
      templateId,
      '1080p',
      30
    );

    // ---- 3️⃣ Build a public URL for the client ----
    const videoUrl = `/generated_videos/${path.basename(outputPath)}`;

3. **Video Generation Trigger** – Clicking “Generate” sends a POST to `/api/generate` with the script content and `templateId`.

5. **Error Handling** –  
   * Unsupported file type → HTTP 415.  
   * Missing `templateId` → HTTP 400.  
   * Internal error → HTTP 500 with generic message.  
6. **Security** – All inputs are sanitized; no path traversal or injection is possible; generated files are stored in a non‑executable temp directory and cleaned up after download.
